service cloud.firestore {
  match /databases/{database}/documents {
    match /matches/{document=**} {
    	allow create: if request.auth != null;
      allow read, update: if request.auth != null && get(resource.data.player1Reference).data.authentication == request.auth.uid;
      allow read, update: if request.auth != null && get(resource.data.player2Reference).data.authentication == request.auth.uid;
    }

    match /questions/{document=**} {
    	allow read, write: if request.auth != null;
    }

    match /users/{document=**} {
    	allow list, get, create: if request.auth != null;
      allow update, delete: if request.auth != null && resource.authentication == request.auth.uid;
    }
  }
}
